### GATHER THE ICONS

```sh
deposit=".assets" && mkdir -p "$deposit"
baseurl="https://github.com/olankens/iconawan/raw/HEAD/source"
for f in {angular,chromium,kotlin,kubernetes,spring}; do
	curl -Lo "$deposit/icon-$f.svg" "$baseurl/$f.svg"
done
```
